	@password nvarchar(200)

	)

	as

	begin

	Insert into UserDetails values (@username, @password)

	End

	End




UPDATE RECORD IN DATABASE

	CREATE PROCEDURE [dbo].[UpdateRecord]

	(	
	
 	@StdId int, 
	@username nvarchar(50),
	@password nvarchar(200)

	)
	as

	begin

	update UserDetails
	set username=@username,
	password=@password
	where Id=@StdId

	end
